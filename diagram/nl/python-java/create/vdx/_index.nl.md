```cs
import  jpype     
import  asposediagram     
jpype.startJVM() 
from asposediagram.api import *
// Maak Diagram klasse-instantie.
diagram = new Diagram();

// Open de eerste pagina van de diagram.
page = diagram.getPages().get(0);

// Tekstvorm toevoegen.
shape = page.addText(1, 1, 1, 1, "Test text", "Calibri", "#a5a5a5", 0.25);

// Sla het Diagram op als .vdx-bestand.
diagram.save("out.vdx",SaveFileFormat.VDX);
```
