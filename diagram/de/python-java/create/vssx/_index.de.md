```cs
import  jpype     
import  asposediagram     
jpype.startJVM() 
from asposediagram.api import *
// Diagram-Klasseninstanz erstellen.
diagram = new Diagram();

// Greifen Sie auf die erste Seite von diagram zu.
page = diagram.getPages().get(0);

// Textform hinzufügen.
shape = page.addText(1, 1, 1, 1, "Test text", "Calibri", "#a5a5a5", 0.25);

// Speichern Sie Diagram als .vssx-Datei.
diagram.save("out.vssx",SaveFileFormat.VSSX);
```
