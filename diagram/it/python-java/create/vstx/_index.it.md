import  jpype     
import  asposediagram     
jpype.startJVM() 
from asposediagram.api import *
// Crea Diagram istanza di classe.
diagram = new Diagram();

// Accedi alla prima pagina di diagram.
page = diagram.getPages().get(0);

// Aggiungi forma del testo.
shape = page.addText(1, 1, 1, 1, "Test text", "Calibri", "#a5a5a5", 0.25);

// Salva il file Diagram come .vstx.
diagram.save("out.vstx",SaveFileFormat.VSTX);
